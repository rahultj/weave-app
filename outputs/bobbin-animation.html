Usage

// Trigger states with class changes
const bobbin = document.getElementById('bobbin');

// When user sends a message
bobbin.className = 'bobbin-container anticipation';
setTimeout(() => bobbin.className = 'bobbin-container diving', 200);
setTimeout(() => bobbin.className = 'bobbin-container searching', 700);

// When response starts streaming
bobbin.className = 'bobbin-container emerging';
setTimeout(() => bobbin.className = 'bobbin-container idle', 500);
